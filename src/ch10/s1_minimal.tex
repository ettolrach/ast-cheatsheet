\wc{Recall:} $\mathbf{m} \leq \mathbf{n} \iff m \leq n$; $\mathbf{m} \cong \mathbf{n} \iff m = n$.
\wde{n.n.} A set is \textit{infinite} if it's not finite.
\wl{Remark 10.1.1 Pigeonhole principle} Let $m, n \in \N$. If $m > n$ then there is no injection $\mathbf{m} \to \mathbf{n}$.
\wde{n.n.} The \textit{cardinality} of a finite set $X$, writen as $|X|$, is the unique natural number $n$ such that $X \cong \mathbf{n}$.
\wl{10.1.2} Let $X$ finite and $Y$ be sets. If $\exists$ an injection $Y \to X$ or a surjection $X \to Y$, then $Y$ is also finite.
\wl{10.1.4} Let $X$ and $Y$ be sets. Then $X \leq Y$ iff $X$ is empty or $\exists$ a surjection $Y \to X$.
\wpr{10.1.5} Let $X$ be a set. These are equivalent: \textbf{(i)} $X$ is infinite; \textbf{(ii)} $\N \leq X$, that is, $\exists$ an injection $\N \to X$; \textbf{(iii)} $\exists$ a surjection $X \to \N$; \textbf{(iv)} $X \cong X + \mathbf{1}$; \textbf{(v)} $\exists$ a non-surjective injection $X \to X$; \textbf{(vi)} $\exists$ a non-injective surjection $X \to X$.
\wc{10.1.8} All subsets of $\N$ are either iso. to $\N$ or finite.
